{\Huge \bfseries \noindent Week 5 Quiz}

\section*{Multiple Choice}

\begin{enumerate}
    \item[1.] [5 points] Consider the following function:
    \[
        f(x) = x\ln(x) + e^x x^2.
    \]
    What is the value of the derivative $f'(x)$ at the point $x = 0$?
    \begin{enumerate}
        \item $f'(0) = -\infty$
        \item $f'(0) = 0$
        \item $f'(0) = -0.5$
        \item $f'(0) = -1$
    \end{enumerate}

    \item[2.] [5 points] What is the $2026$'th derivative of the function $\cos(-2x)$?
    \begin{enumerate}
        \item ${(-2)}^{2026}\sin(x)$
        \item $-2\cdot{(-1)}^{2026}\sin(-2x)$
        \item $-{(2026)}^{-2}\cos(-2x)$
        \item $-{(2)}^{2026}\cos(-2x)$
    \end{enumerate}
\end{enumerate}

\section*{Written Response}
