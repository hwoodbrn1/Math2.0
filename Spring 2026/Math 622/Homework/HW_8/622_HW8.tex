    \item[1.] (a.) Let 
    \begin{gather*}
        g(\theta, \phi) = \left((\cos\phi + 2)\cos\theta,(\cos\phi + 2)\sin\theta, \sin\phi\right)\\
        \ \\
        \omega = y dz\wedge dx
    \end{gather*}

    To compute $g^*\omega$, we can use the formula 
    \[
        F^*\left(\sum_I \omega_I dy^{I_1}\wedge \cdots\wedge dy^{I_k}\right) = \sum_I
        (\omega_I \circ F)d(y^{I_1}\circ F) \wedge \cdots \wedge d(y^{I_k}\circ F).
    \]

    Then we have
    \begin{align*}
        &g^*\omega = (y \circ g)d(z\circ g)\wedge(x \circ g)\\
        &= ((\cos\phi + 2)\sin\theta)d(\sin\phi)\wedge d((\cos\phi +2)\cos\theta)\\
    \end{align*}

    For the term $d((\cos\phi + 2)\cos\theta)$, we use the product rule:
    \begin{align*}
        &d((\cos\phi + 2)\cos\theta) = d(\cos\phi + 2)\cos\theta + (\cos\phi + 2)d\cos\theta\\
        & = -\sin\phi\cos\theta d\phi - (\cos\phi + 2)\sin\theta d\theta.
    \end{align*}

    Putting this together, we have
    \begin{align*}
        &g^*\omega = ((\cos\phi + 2)\sin\theta)\cos\phi d\phi\wedge (-\sin\phi\cos\theta d\phi - (\cos\phi + 2)\sin\theta d\theta)\\
        &= -((\cos\phi + 2)\sin\theta)\cos\phi d\phi \wedge (\cos\phi + 2)\sin\theta d\theta\\
        &= -(\cos\phi + 2)^2 \sin^2\theta\cos\phi d\phi \wedge d\theta,
    \end{align*}
    since $d\phi \wedge d\phi = 0$.

    To compute $d\omega$, we will use the derivation property as well as the formula
    \[
        d(f dz \wedge dx) = \frac{\partial f}{\partial x} dy \wedge dz \wedge dx,
    \]
    which is immediate from the definition of the exterior derivative, since only the $\frac{\partial f}{\partial y}$ term survives
    in the sum. Then it is immediate that 
    \[
        d(y dz\wedge dx) = d(y dz) \wedge dx - y dz \wedge d(dx) = dy \wedge dz \wedge dx.
    \]

    Since $d\omega$ is a three form, and the domain of $g$ is a two dimensional manifold, the pullback $g^*(d\omega) = 0$. Likewise,
    $d(g^*\omega) = 0$ as well, since $g^*\omega$ is a top form, so its exterior derivative must be zero. 

    (b.) Next we let 
    \begin{gather*}
        g(u,v) = (u, v, \sqrt{1 - u^2 - v^2})\\
        \omega = \frac{x dy \wedge dz + y dz \wedge dx + z dx \wedge dy}{(x^2 + y^2 + z^2)^{3/2}}.
    \end{gather*}

    Let $r = (x^2 + y^2 + z^2)$. We first compute $d\omega$. For 
    \[
        \omega = f dy \wedge dz + g dz \wedge dx + h dx \wedge dy,
    \]
    we have 
    \[
        d\omega = (\frac{\partial f}{\partial x} + \frac{\partial g}{\partial y} + \frac{\partial h}{\partial z}) dx \wedge dy \wedge dz.
    \]
    In our case,
    \[
        \frac{\partial f}{\partial x} = \frac{\partial}{\partial x}\left(\frac{x}{r^{3/2}}\right) = \frac{r - 3x^2}{r^{5/2}}.
    \]
    Then
    \[
        d\omega = \frac{3r - 3x^2 - 3y^2 - 3z^2}{r^{5/2}}dx \wedge dy\wedge dz = \frac{3r - 3r}{r^{5/2}}dx \wedge dy \wedge dz = 0.
    \]

    Let $r = \sqrt{1 - u^2 - v^2}$. For $g^*\omega$, we calculate
    \begin{align*}
        &d(x\circ g) = d(u) = du\\
        &d(y \circ g) = dv\\
        &d(z \circ g) = d(r) = \frac{-2u du - 2v dv}{2r} = s
    \end{align*}
    Then 
    \[
        g^*\omega = u dv \wedge s + v s \wedge du + r du \wedge dv = \frac{u^2 + v^2}{r} du \wedge dv + r du \wedge dv = \frac{1}{r}du \wedge dv
    \]

    after cancelling $du \wedge du$ and $dv \wedge dv$ terms. Again, $g^*(d\omega) = d(g^*\omega) = 0$. 

    \item[2.] Let $M = \RR^3$ with coordinates $(x,y,z)$ and $g$ the standard flat metric. Let $X = P \frac{\partial}{\partial x} + Q 
    \frac{\partial}{\partial y} + R \frac{\partial}{\partial z}$. 

    The Hodge star operator $*$ acts on one forms by 
    \[
        *(a dx + b dy + c dz) = a dy \wedge dz + b dz \wedge dx + c dx \wedge dy,
    \]
    where we have chosen $dx \wedge dy \wedge dz$ to be positively oriented, and it is its own inverse. 

    The operator $W$ is the map
    \[
        W(a \frac{\partial}{\partial x} + b \frac{\partial}{\partial y} + c \frac{\partial}{\partial z}) = a dx + b dy + c dz,
    \]
    since the action of $a dx + b dy + c dz$ on an arbitrary vector field is the same as the euclidean inner product of these
    two vector fields. The inverse is clear.

    (a.) We compute 
    \begin{align*}
        * \circ d \circ * \circ W(X) & = * \circ d \circ *(P dx + Q dy + R dz)\\
        &= *\circ d(P dy  \wedge dz + Q dz \wedge dx + R dx \wedge dy)\\
        &= *\left(\left(\frac{\partial P}{\partial x} + \frac{\partial Q}{\partial y} + \frac{\partial R}{\partial z}\right)dx \wedge dy \wedge dz\right)\\
        &= \left(\frac{\partial P}{\partial x} + \frac{\partial Q}{\partial y} + \frac{\partial R}{\partial z}\right)
    \end{align*}

    (b.) Again,
    \begin{align*}
        W^{-1}\circ * \circ d \circ W(X) & = W^{-1}\circ * \circ d(P dx + Q dy + Rdz)\\
        & = W^{-1}\circ *\left(\frac{\partial P}{\partial y}dy \wedge dx  +\frac{\partial P}{\partial z} dz \wedge dx 
        + \frac{\partial Q}{\partial x} dx \wedge dy + \frac{\partial Q}{\partial z} dz \wedge dy + \frac{\partial R}{\partial x} dx \wedge dz
        + \frac{\partial R}{\partial y} dy \wedge dz\right)\\
        & = W^{-1}\circ *\left(\left(\frac{\partial Q}{\partial x} - \frac{\partial P}{\partial y}\right) dx \wedge dy
        + \left(\frac{\partial R}{\partial y} - \frac{\partial Q}{\partial z}\right)dy \wedge dz
        + \left(\frac{\partial P}{\partial z} - \frac{\partial R}{\partial x}\right)dz \wedge dx\right)\\
        & = W^{-1}\left(\left(\frac{\partial Q}{\partial x} - \frac{\partial P}{\partial y}\right) dz
        + \left(\frac{\partial R}{\partial y} - \frac{\partial Q}{\partial z}\right)dx
        + \left(\frac{\partial P}{\partial z} - \frac{\partial R}{\partial x}\right)dy\right)\\
        & = \left(\frac{\partial R}{\partial y} - \frac{\partial Q}{\partial z}\right)\frac{\partial}{\partial x}
        + \left(\frac{\partial P}{\partial z} - \frac{\partial R}{\partial x}\right)\frac{\partial}{\partial y}
        + \left(\frac{\partial Q}{\partial x} - \frac{\partial P}{\partial y}\right) \frac{\partial}{\partial z}.
    \end{align*}

    \item[3.] Let $\omega$ be an $r$-form. We will prove that 
    \[
        d\omega(X_0, X_1, \dots, X_r) = \sum_{i = 0}^r (-1)^i X_i(\omega(X_0, \dots, \hat{X_i}, \dots, X_r))
        + \sum_{0 \leq i < j \leq r} (-1)^{i + j}\omega([X_i, X_j], X_0, \dots, \hat{X_i}, \dots, \hat{X_j}, \dots, X_r)
    \]
    using induction on the order of $\omega$. 

    We first prove for $r = 0$. Let $f$ be a $0$-form, or a function. Then
    \[  
        df(X_0) = X_0(f).
    \]
    This is the formula above, since in this case the second sum is empty.

    For induction, suppose it has been proven for $r-1$-forms. 

    We have Cartan's magic formula,
    \[
        \mathcal{L}_{X}(\omega) = i_{X}(d\omega) + d(i_{X}\omega),
    \]
    which tells us that 
    \begin{gather*}
        d\omega(X_0, \dots, X_r) = (-1)^r i_{X_r}\omega(X_0, \dots, X_{r-1}) = 
        \left((-1)^r \mathcal{L}_{X_r}\omega - (-1)^r d(i_{X_r}\omega)\right)(X_0, \dots, X_{r-1}).
    \end{gather*}

    Starting from the formula we derived in the last homework,
    \begin{align*}
        (\mathcal{L}_{X_r}\omega)(X_0, \dots, X_{r-1}) & = X_r(\omega(X_0, \dots, X_{r-1})) - 
        \omega([X_r, X_0], X_1, \dots, X_{r-1}) - \cdots - \omega(X_0, \dots, [X_r, X_{r-1}])\\
        & = X_r(\omega(X_0, \dots, X_{r-1})) - \sum_{i = 0}^{r-1}(-1)^i\omega([X_r, X_i], \dots, \hat{X_i},\dots, X_{r-1})\\
        & = X_r(\omega(X_0, \dots, X_{r-1})) + \sum_{i = 0}^{r-1}(-r)^i\omega([X_i, X_r], \dots, \hat{X_i}, \dots, X_{r-1}, \hat{X_r}).
    \end{align*}

    By the induction hypothesis, 
    \begin{align*}
        d(i_{X_r}\omega) &= \sum_{i = 0}^{r-1} (-1)^i X_i(i_{X_r}\omega(X_0, \dots, \hat{X_i}, \dots, X_{r-1}))\\
        & + \sum_{0 \leq i < j \leq r-1}(-1)^{i + j}i_{X_r}\omega([X_i, X_j], \dots, \hat{X_i}, \dots, \hat{X_j}, \dots, X_{r-1})\\
        & = \sum_{i = 0}^{r-1} (-1)^i X_i(\omega(X_r, X_0, \dots, \hat{X_i}, \dots, X_{r-1}))\\
        & + \sum_{0 \leq i < j \leq r-1}(-1)^{i + j}\omega(X_r, [X_i, X_j], \dots, \hat{X_i}, \dots, \hat{X_j}, \dots, X_{r-1})\\
        & = (-1)^r \sum_{i = 0}^{r-1} (-1)^i X_i(\omega(X_0, \dots, \hat{X_i}, \dots, X_{r}))\\
        & + (-1)^r\sum_{0 \leq i < j \leq r-1}(-1)^{i + j}\omega([X_i, X_j], \dots, \hat{X_i}, \dots, \hat{X_j}, \dots, X_{r}).
    \end{align*}

    Putting this all together,
    \begin{align*}
        d\omega(X_0, \dots, X_r) 
        & = (-1)^r X_r(\omega(X_0, \dots, X_{r-1})) + \sum_{i = 0}^{r-1}(-1)^{i + r}\omega([X_i, X_r], \dots, \hat{X_i}, \dots, X_{r-1}, \hat{X_r})\\
        & + \sum_{i = 0}^{r-1} (-1)^i X_i(\omega(X_0, \dots, \hat{X_i}, \dots, X_{r}))
         + \sum_{0 \leq i < j \leq r-1}(-1)^{i + j}\omega([X_i, X_j], \dots, \hat{X_i}, \dots, \hat{X_j}, \dots, X_{r})\\
         & = \sum_{i = 0}^r (-1)^i X_i(\omega(X_0, \dots, \hat{X_i}, \dots, X_r))
        + \sum_{0 \leq i < j \leq r} (-1)^{i + j}\omega([X_i, X_j], X_0, \dots, \hat{X_i}, \dots, \hat{X_j}, \dots, X_r).
    \end{align*}

    \item[4.] Let $(X_1, \dots, X_n)$ be a frame on a manifold $M$ and $(\omega^1, \dots, \omega^n)$ the dual coframe. 
    (a.) Since $[X_i, X_j]$ is a smooth vector field for every $i, j$, it can be expressed at each point in the basis 
    given by the frame. Then there are coefficients $c_{ij}^k$, $1 \leq i,j,k \leq n$ such that 
    \[
        [X_i, X_j] = \sum_{k = 1}^n c_{ij}^k X_k.
    \]
    Using the identity $[X_i, X_j] = -[X_j, X_i]$, we see that 
    \[
        \sum_{k = 1}^n c_{ij}^k X_k = \sum_{k = 1}^n c_{ji}^k X_k,
    \]
    so that
    \[
        \sum_{k = 1}^n (c_{ij}^k - c_{ji}^k) X_k = 0,
    \]
    and thus $c_{ij}^k = c_{ji}^k$ by linear independence. 

    To prove smoothness of the coefficients, we use the fact that the pairing of a one form with a vector field results in 
    a smooth function on $M$. In general, if $Y = \sum_{i = 1}^n y^i X_i$ is a vector field, we apply the covector field
    $\omega^i$:
    \[
        \omega^i(Y) = y_i.
    \]
    Then the coefficients of $Y$ are smooth functions. This applies to the Lie bracket above. 

    In the dual case, we have a basis for $\Lambda^2 M$ given by $\omega^i \wedge \omega^j$ for $1 \leq i < j \leq n$. Since
    $d\omega^k$ is a $2$-form, we can write
    \[
        d\omega^k = \sum_{1 \leq i < j \leq n} \tilde{e}_{ij}^k \omega^i \wedge \omega^j.
    \]
    Then let $e_{ij}^k = \frac{\tilde{e}_{ij}^k}{2}$ for $i < j$ and $e_{ij}^k = -\frac{\tilde{e}_{ji}^k}{2}$ for $j < i$. 
    Note the diagonals of both matrices are zero. Then 
    \begin{align*}
        & \sum_{1 \leq i, j \leq n} e_{ij}^k \omega^i \wedge \omega^j 
        = \sum_{1 \leq i < j \leq n}(e_{ij}^k - e_{ji}^k)\omega^i \wedge \omega^j\\
        & = \sum_{1 \leq i < j \leq n}\left(\frac{\tilde{e}_{ij}^k + \tilde{e}_{ij}^k}{2}\right)
        = \sum_{1 \leq i < j \leq n}\tilde{e}_{ij}^k \omega^i \wedge \omega^j = d\omega^k.
    \end{align*}

    We also have $e_{ij}^k = - e_{ji}^k$ by definition. The coefficients are smooth since 
    $d\omega^k(X_i, X_j) = e_{ij}^k$, and a smooth 2-form applied to two vector fields gives a smooth function. 

    (b.) We use the formula from (3.):
    \[
        d\omega^k(X_i, X_j) = X_i(\omega^k(X_j)) - X_j(\omega^k(X_i)) - \omega^k([X_i, X_j]).
    \]
    In this case, since $\omega^k(X_j)$ takes the value zero or one, the first two terms are zero. For the last term,
    \[
        \omega^k([X_i, X_j]) = \omega^k\left(\sum_{k = 1}^n c_{ij}^k X_k\right) = c_{ij}^k.
    \]
    In part (a.) we showed that $d\omega^k(X_i, X_j) = e_{ij}^k$. Then $e_{ij}^k = c_{ij}^k$.

    \item[5.] Let $\omega$ be a closed non-degenerate $2$ form on $M$. 
    (a.) Given a coordinate neighborhood, any 2-form is represented 
    by a skew-symmetric matrix, say $A$. In detail, we let
    \[
        A_{ij} = \omega\left(\frac{\partial}{\partial x_i}, \frac{\partial}{\partial x_j}\right).
    \]
    It is clear that $A$ is skew symmetric. We have 
    \[
        \omega(X, Y) = Y^t A X
    \]

    for vector fields $X$ and $Y$, where the transpose is only to describe the action of $A$. 
    The non-degeneracy condition means that if $AX = 0$, then $X$ is zero. This means $A$ is an
    invertible matrix. 

    Let $h$ be a smooth function. We want to find a vector field $H$ such that $i_{H}\omega = -dh$. In 
    other words, we want to solve $i_H\omega (X) = X^t A H = -dh(X)$ for $H$, where $X$ is an arbitrary vector field. 
    But we can do this, since $A$ is invertible. Let $X = \frac{\partial}{\partial x_i}$ be a vector field in local
    coordinates for a fixed $i$. Then 
    \begin{align*}
        X^t A H = (A H)_i dx^i = -dh = -\sum_{i = 1}^n \frac{\partial h}{\partial x_i} dx^i = -\frac{\partial h}{\partial x_i}dx^i,
    \end{align*}

    and we can take $H_i = -{(A^{-1})}_i^j(\frac{\partial h}{\partial x_j})$.

    (b.) We use Cartan's magic formula: $\mathcal{L}_{H}\omega = i_H(d\omega) + d(i_H\omega)$. Here $d\omega = 0$ since $\omega$
    is closed, and the second term is zero since $i_H\omega$ is exact. Then $\mathcal{L}_H \omega = 0$. 

    Moreover, the Lie derivative is the derivative of the pushforward $(e^{tH})^*\omega$ at time $t = 0$, but we 
    can choose any $t$ which our flow is defined to take our derivative, since the derivative 
    at $t = s$ is the same as taking the derivative at $t = 0$ of the related flow which is $s$ ahead of the original
    flow. But these determine the same vector field. Then the pushforward is constant for all time. But we know 
    that $(e^{0})^*\omega = \omega$, so $(e^{iH})^*\omega = \omega$ for all $t$. 

    (c.) (i.) Let $\lambda = (q,p)$ and $v \in T_\lambda(T^*M)$. We define 
    \[
        s(\lambda)(v) = p(d\pi(v)).
    \]
    We can write $v = \sum a_i \frac{\partial}{\partial x_i}|_q + \sum b_i \frac{\partial}{\partial y_j}|_p$, so
    that $d\pi(v) = \sum a_i \frac{\partial}{\partial x_i}$. 

    We know that $s$ has some coefficients $s_j, r_j$ such that 
    \[
        s = \sum s_j dx^i + \sum r_j dy^i.
    \]
    If we let $v = \frac{\partial}{\partial x_i}$, we see that 
    \[
        s_i = p(d\pi(v)) = y_i.
    \]
    It is also clear that the $r_j$'s must be zero because of the projection. 

    (ii.) Taking the differential, we see that
    \[
        ds = \sum dy_i \wedge dx_i.
    \]

    Let $X = \sum a_i \frac{\partial}{\partial x_i} + \sum b_i \frac{\partial}{\partial y_i}$. 
    Suppose $i_X \omega = 0$. We can represent $i_X \omega$ in terms of the coordinate basis 
    for $T^*M$. We have 
    \[
        i_X\omega(\frac{\partial}{\partial x_i}) = \sum b_i dx^i - \sum a_i dy^i.
    \]
    Then if $i_X \omega = 0$, each of the coefficients of $X$ must be zero by linear independence. 

\end{enumerate}
